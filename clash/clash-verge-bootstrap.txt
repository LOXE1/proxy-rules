// 保存为 TXT 只是便于查看；内容仍应粘贴到 Clash Verge Rev 的“全局扩展脚本”。

function main(config, profileName) {
  const TEST_URL = "https://www.gstatic.com/generate_204";
  const INVALID_NODE_FILTER = "(?i)(剩余|流量|套餐|到期|有效期|官网|网址|重置|更新订阅|traffic|expire|website|reset|update)";

  if (typeof config["tcp-concurrent"] === "undefined") config["tcp-concurrent"] = true;
  if (typeof config["find-process-mode"] === "undefined") config["find-process-mode"] = "strict";

  const REGIONS = {
    HK: { regionGroup: "🇭🇰 香港｜自动或手动", autoGroup: "♻️ 香港｜最低延时", filter: "(?i)(🇭🇰|香港|Hong[ ._-]*Kong|\\bHK\\b)" },
    TW: { regionGroup: "🇹🇼 台湾｜自动或手动", autoGroup: "♻️ 台湾｜最低延时", filter: "(?i)(🇹🇼|台湾|台灣|Taiwan|Taipei|台北|\\bTW\\b)" },
    JP: { regionGroup: "🇯🇵 日本｜自动或手动", autoGroup: "♻️ 日本｜最低延时", filter: "(?i)(🇯🇵|日本|Japan|Tokyo|Osaka|东京|東京|大阪|\\bJP\\b)" },
    SG: { regionGroup: "🇸🇬 新加坡｜自动或手动", autoGroup: "♻️ 新加坡｜最低延时", filter: "(?i)(🇸🇬|新加坡|狮城|獅城|Singapore|\\bSG\\b)" },
    US: { regionGroup: "🇺🇸 美国｜自动或手动", autoGroup: "♻️ 美国｜最低延时", filter: "(?i)(🇺🇸|美国|美國|United[ ._-]*States|America|\\bUSA\\b|\\bUS\\b|Los[ ._-]*Angeles|New[ ._-]*York|San[ ._-]*Jose|Seattle|Dallas|Chicago|Miami|Virginia|Phoenix|Portland|Atlanta|\\bLA\\b|\\bNY\\b)" },
    VN: { regionGroup: "🇻🇳 越南｜自动或手动", autoGroup: "♻️ 越南｜最低延时", filter: "(?i)(🇻🇳|越南|Vietnam|Ho[ ._-]*Chi[ ._-]*Minh|HoChiMinh|Hanoi|河内|河內|胡志明|\\bHCM\\b|\\bVN\\b)" },
    TH: { regionGroup: "🇹🇭 泰国｜自动或手动", autoGroup: "♻️ 泰国｜最低延时", filter: "(?i)(🇹🇭|泰国|泰國|Thailand|Bangkok|曼谷|\\bBKK\\b|\\bTH\\b)" },
    ID: { regionGroup: "🇮🇩 印尼｜自动或手动", autoGroup: "♻️ 印尼｜最低延时", filter: "(?i)(🇮🇩|印尼|印度尼西亚|印度尼西亞|Indonesia|Jakarta|雅加达|雅加達|\\bJKT\\b|\\bID\\b)" },
    IN: { regionGroup: "🇮🇳 印度｜自动或手动", autoGroup: "♻️ 印度｜最低延时", filter: "(?i)(🇮🇳|印度|India|Mumbai|Bombay|Delhi|Bangalore|Bengaluru|Chennai|Kolkata|孟买|孟買|新德里|\\bIN\\b)" }
  };

  const APP_GROUPS = { OPENAI: "OpenAI", GEMINI: "Gemini", CLAUDE: "Claude", COPILOT: "Copilot", DISCORD: "Discord", TELEGRAM: "Telegram", TIKTOK: "TikTok" };
  const FIXED_GROUPS = { VN: "🇻🇳 越南｜端口 10001", TH: "🇹🇭 泰国｜端口 10002", US: "🇺🇸 美国｜端口 10003" };

  function makeAutoGroup(region) {
    return { name: region.autoGroup, type: "url-test", url: TEST_URL, interval: 300, timeout: 5000, tolerance: 50, lazy: true, "max-failed-times": 3, "expected-status": 204, "include-all": true, filter: region.filter, "exclude-filter": INVALID_NODE_FILTER, "empty-fallback": "DIRECT", hidden: true };
  }
  function makeRegionGroup(region) {
    return { name: region.regionGroup, type: "select", proxies: [region.autoGroup], "include-all": true, filter: region.filter, "exclude-filter": INVALID_NODE_FILTER, "empty-fallback": "DIRECT" };
  }
  function makeFixedPortGroup(name, region) {
    return { name: name, type: "select", "include-all": true, filter: region.filter, "exclude-filter": INVALID_NODE_FILTER, "empty-fallback": "REJECT" };
  }
  function makeAppGroup(name, regionKeys) {
    return { name: name, type: "select", proxies: regionKeys.map(function (key) { return REGIONS[key].regionGroup; }) };
  }

  const fixedPortGroups = [makeFixedPortGroup(FIXED_GROUPS.VN, REGIONS.VN), makeFixedPortGroup(FIXED_GROUPS.TH, REGIONS.TH), makeFixedPortGroup(FIXED_GROUPS.US, REGIONS.US)];
  const applicationGroups = [
    makeAppGroup(APP_GROUPS.OPENAI, ["SG", "US", "JP", "TW", "HK"]),
    makeAppGroup(APP_GROUPS.GEMINI, ["US", "SG", "JP", "TW", "HK"]),
    makeAppGroup(APP_GROUPS.CLAUDE, ["US", "SG", "JP", "TW", "HK"]),
    makeAppGroup(APP_GROUPS.COPILOT, ["US", "SG", "JP", "TW", "HK"]),
    makeAppGroup(APP_GROUPS.DISCORD, ["SG", "HK", "JP", "US", "TW"]),
    makeAppGroup(APP_GROUPS.TELEGRAM, ["SG", "HK", "JP", "US", "TW", "TH", "VN", "ID", "IN"]),
    makeAppGroup(APP_GROUPS.TIKTOK, ["ID", "IN", "TH", "VN", "US", "SG", "JP", "TW", "HK"])
  ];
  const regionGroups = Object.keys(REGIONS).map(function (key) { return makeRegionGroup(REGIONS[key]); });
  const autoGroups = Object.keys(REGIONS).map(function (key) { return makeAutoGroup(REGIONS[key]); });
  const managedGroups = fixedPortGroups.concat(applicationGroups, regionGroups, autoGroups);
  const managedGroupNames = managedGroups.map(function (group) { return group.name; });
  const oldGroups = Array.isArray(config["proxy-groups"]) ? config["proxy-groups"] : [];
  config["proxy-groups"] = managedGroups.concat(oldGroups.filter(function (group) { return group && managedGroupNames.indexOf(group.name) === -1; }));

  const fixedListeners = [
    { name: "env-vn-10001", type: "mixed", listen: "127.0.0.1", port: 10001, proxy: FIXED_GROUPS.VN, udp: true },
    { name: "env-th-10002", type: "mixed", listen: "127.0.0.1", port: 10002, proxy: FIXED_GROUPS.TH, udp: true },
    { name: "env-us-10003", type: "mixed", listen: "127.0.0.1", port: 10003, proxy: FIXED_GROUPS.US, udp: true }
  ];
  const fixedListenerNames = fixedListeners.map(function (listener) { return listener.name; });
  const oldListeners = Array.isArray(config.listeners) ? config.listeners : [];
  config.listeners = oldListeners.filter(function (listener) { return listener && fixedListenerNames.indexOf(listener.name) === -1; }).concat(fixedListeners);

  const RAW_BASE = "https://raw.githubusercontent.com/LOXE1/proxy-rules/main/clash/rules/";
  const managedProviders = {
    "pr-openai": { type: "http", behavior: "classical", format: "yaml", url: RAW_BASE + "openai.yaml", path: "./ruleset/proxy-rules/openai.yaml", interval: 86400 },
    "pr-gemini": { type: "http", behavior: "classical", format: "yaml", url: RAW_BASE + "gemini.yaml", path: "./ruleset/proxy-rules/gemini.yaml", interval: 86400 },
    "pr-claude": { type: "http", behavior: "classical", format: "yaml", url: RAW_BASE + "claude.yaml", path: "./ruleset/proxy-rules/claude.yaml", interval: 86400 },
    "pr-copilot": { type: "http", behavior: "classical", format: "yaml", url: RAW_BASE + "copilot.yaml", path: "./ruleset/proxy-rules/copilot.yaml", interval: 86400 },
    "pr-discord": { type: "http", behavior: "classical", format: "yaml", url: RAW_BASE + "discord.yaml", path: "./ruleset/proxy-rules/discord.yaml", interval: 86400 },
    "pr-telegram": { type: "http", behavior: "classical", format: "yaml", url: RAW_BASE + "telegram.yaml", path: "./ruleset/proxy-rules/telegram.yaml", interval: 86400 },
    "pr-tiktok": { type: "http", behavior: "classical", format: "yaml", url: RAW_BASE + "tiktok.yaml", path: "./ruleset/proxy-rules/tiktok.yaml", interval: 86400 }
  };
  const oldProviders = config["rule-providers"] || {};
  const cleanedProviders = {};
  Object.keys(oldProviders).forEach(function (name) {
    if (name.indexOf("pr-") !== 0 && name.indexOf("cv-ai-") !== 0 && name.indexOf("cv-app-") !== 0) cleanedProviders[name] = oldProviders[name];
  });
  config["rule-providers"] = Object.assign({}, cleanedProviders, managedProviders);

  const managedRules = [
    "PROCESS-PATH,/Applications/Discord.app/Contents/MacOS/Discord," + APP_GROUPS.DISCORD,
    "PROCESS-NAME,Discord," + APP_GROUPS.DISCORD,
    "PROCESS-NAME,Discord Helper," + APP_GROUPS.DISCORD,
    "PROCESS-NAME,Discord Helper (Renderer)," + APP_GROUPS.DISCORD,
    "PROCESS-PATH,/Applications/Telegram.app/Contents/MacOS/Telegram," + APP_GROUPS.TELEGRAM,
    "PROCESS-NAME,Telegram," + APP_GROUPS.TELEGRAM,
    "PROCESS-NAME,telegram-desktop," + APP_GROUPS.TELEGRAM,
    "RULE-SET,pr-openai," + APP_GROUPS.OPENAI,
    "RULE-SET,pr-gemini," + APP_GROUPS.GEMINI,
    "RULE-SET,pr-claude," + APP_GROUPS.CLAUDE,
    "RULE-SET,pr-copilot," + APP_GROUPS.COPILOT,
    "RULE-SET,pr-discord," + APP_GROUPS.DISCORD,
    "RULE-SET,pr-telegram," + APP_GROUPS.TELEGRAM,
    "RULE-SET,pr-tiktok," + APP_GROUPS.TIKTOK
  ];

  const managedPolicies = Object.keys(APP_GROUPS).map(function (key) { return APP_GROUPS[key]; });
  const oldRules = Array.isArray(config.rules) ? config.rules : [];
  const cleanedRules = oldRules.filter(function (rule) {
    if (typeof rule !== "string") return true;
    if (rule.indexOf("RULE-SET,pr-") === 0 || rule.indexOf("RULE-SET,cv-ai-") === 0 || rule.indexOf("RULE-SET,cv-app-") === 0) return false;
    const parts = rule.split(",");
    if (parts.length >= 3) {
      const policy = parts[parts.length - 1] === "no-resolve" ? parts[parts.length - 2] : parts[parts.length - 1];
      if (managedPolicies.indexOf(policy) !== -1) return false;
    }
    return true;
  });

  config.rules = managedRules.concat(cleanedRules);
  return config;
}
